# Create namespace
kubectl create namespace monitoring

# Add Helm repo
helm repo add prometheus-community https://prometheus-community.github.io/helm-charts
helm repo update

# Install Prometheus + Grafana (kube-prometheus-stack)
helm install prometheus-stack prometheus-community/kube-prometheus-stack -n monitoring

# Verify pods
kubectl get pods -n monitoring

# Get Grafana admin password
kubectl get secret prometheus-stack-grafana -n monitoring -o jsonpath="{.data.admin-password}" | base64 --decode ; echo

# Access Grafana
minikube service prometheus-stack-grafana -n monitoring --url

# Access Prometheus
minikube service prometheus-stack-kube-prom-prometheus -n monitoring --url
